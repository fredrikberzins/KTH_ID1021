public class Stack {
    private static int stackpos = 0;
    private int value = 0;

    public void push(int args) {
        stackpos++;
        value = args;
    }
    public int pop() {
        int ReturnValue = value;
        stackpos--;
        value = 0;
        return ReturnValue;
    }
    public int peek() {
        return value;
    }
}
